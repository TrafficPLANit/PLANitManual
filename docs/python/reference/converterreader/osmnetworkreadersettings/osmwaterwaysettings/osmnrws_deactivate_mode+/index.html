.deactivate_osm_water_mode(+)

OsmWaterwaySettings method

Description

Deactivate chosen OSM water mode (that is currently activated and supported).

Signature

.deactivate_osm_water_mode(mode: str)

with

Parameter Type Unit Default Description
mode str None N/A OSM water mode that is to be deactivated

Return type

N/A

Example 1

from planit import *

# create a network converter
planit_instance = Planit()
network_converter = planit_instance.converter_factory.create(ConverterType.NETWORK)

# Open Street Map (OSM) network reader        
osm_reader = network_converter.create_reader(NetworkReaderType.OSM, "<country_name>")
osm_reader.settings.set_input_file("<path_to_input_file>")

# deactivate ferry from the water modes to parse
osm_reader.settings.activate_waterway_parser(True)
osm_reader.settings.waterway_settings.deactivate_osm_water_mode("ferry")
